## Install

```
$ npm install --save random-uuid 

## Usage

```js
var randomUuid = require('random-uuid');

// API
// - randomUuid([options]);

randomUuid();     
// => 0616f498-a5fe-4c10-b898-da3d1a22f11d 
randomUuid({ prefix: 'r-' }); 
// => r-a4474ce4-f8c5-454c-82b6-8a206a7b4053
```
